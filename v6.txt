    <script>
        // Framer-specific interactions
        document.addEventListener('DOMContentLoaded', function() {
            // Mobile Menu Toggle
            const menuToggle = document.getElementById('framer-menu-toggle');
            const closeMenu = document.getElementById('framer-close-menu');
            const mobileMenu = document.getElementById('framer-mobile-menu');
            const mobileLinks = document.querySelectorAll('.framer-mobile-link');
            
            menuToggle.addEventListener('click', () => {
                mobileMenu.classList.remove('hidden');
                mobileMenu.classList.add('flex');
            });
            
            closeMenu.addEventListener('click', () => {
                mobileMenu.classList.remove('flex');
                mobileMenu.classList.add('hidden');
            });
            
            mobileLinks.forEach(link => {
                link.addEventListener('click', () => {
                    mobileMenu.classList.remove('flex');
                    mobileMenu.classList.add('hidden');
                    
                    // Scroll to section
                    const targetId = link.getAttribute('data-target');
                    scrollToSection(targetId);
                });
            });
            
            // Scroll Reveal Animation
            const scrollRevealElements = document.querySelectorAll('.framer-scroll-reveal');
            
            const checkScroll = () => {
                scrollRevealElements.forEach(element => {
                    const elementTop = element.getBoundingClientRect().top;
                    const windowHeight = window.innerHeight;
                    
                    if (elementTop < windowHeight * 0.85) {
                        element.classList.add('active');
                    }
                });
                
                // Animate skill bars when in view
                const skillSection = document.getElementById('skills');
                if (skillSection) {
                    const sectionTop = skillSection.getBoundingClientRect().top;
                    const windowHeight = window.innerHeight;
                    
                    if (sectionTop < windowHeight * 0.85) {
                        animateSkillBars();
                    }
                }
            };
            
            // Animate skill bars
            const animateSkillBars = () => {
                const skillItems = document.querySelectorAll('.skill-item');
                
                skillItems.forEach((item, index) => {
                    const percentage = item.getAttribute('data-percentage');
                    const progressFill = item.querySelector('.progress-fill');
                    
                    setTimeout(() => {
                        progressFill.style.width = percentage + '%';
                    }, index * 200);
                });
            };
            
            window.addEventListener('scroll', checkScroll);
            window.addEventListener('load', checkScroll);
            
            // Framer Accordion
            const accordionItems = document.querySelectorAll('.framer-accordion');
            
            accordionItems.forEach(item => {
                const header = item.querySelector('.framer-accordion-header');
                
                header.addEventListener('click', () => {
                    // Close all other accordions
                    accordionItems.forEach(otherItem => {
                        if (otherItem !== item && otherItem.classList.contains('active')) {
                            otherItem.classList.remove('active');
                        }
                    });
                    
                    // Toggle current accordion
                    item.classList.toggle('active');
                });
            });
            
            // Form Submission
            const contactForm = document.getElementById('framer-contact-form');
            
            contactForm.addEventListener('submit', (e) => {
                e.preventDefault();
